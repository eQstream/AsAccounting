cmake_minimum_required(VERSION 3.15)
project(AsAccounting VERSION 1.0.0 LANGUAGES CXX)

set(SOURCES
    main.cpp
    Mesto.cpp
)

add_executable(asaccounting ${SOURCES})
target_include_directories(asaccounting PRIVATE ${CMAKE_CURRENT_SOURCE_DIR})
target_compile_features(asaccounting PRIVATE cxx_std_17)

# Для MinGW — установить флаг UTF-8
if(MINGW)
    target_compile_options(asaccounting PRIVATE -fexec-charset=UTF-8)
endif()
